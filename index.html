function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1"); // Add headers if the sheet is empty if (sheet.getLastRow() === 0) { sheet.appendRow(["Name", "Age", "Gender","Assembly","Mandal","Village","Caste","Contact"]); } const name = e.parameter.name; const age = e.parameter.age; const gender = e.parameter.gender; const assembly = e.parameter.assembly ; const mandal= e.parameter.mandal; const village = e.parameter.village; const caste = e.parameter.caste; const contact = e.parameter.contact; sheet.appendRow([name, age, gender, assembly,mandal,village,caste, contact]); return ContentService .createTextOutput("Success") .setMimeType(ContentService.MimeType.TEXT); }
